# 09 - 管理器模块

## 模块概述

**包路径**: `com.codingapi.flow.manager`

管理器模块封装了各类业务管理功能，协调节点、动作、策略之间的交互。

## 管理器列表

| 管理器 | 说明 |
|--------|------|
| `ActionManager` | 节点动作管理器 |
| `FlowNodeManager` | 流程节点管理器 |
| `OperatorManager` | 操作者管理器 |
| `WorkflowStrategyManager` | 工作流策略管理器 |

## ActionManager

```java
package com.codingapi.flow.manager;

/**
 * 节点动作管理器
 * 负责管理节点的所有动作并进行会话验证
 */
public class ActionManager {

    /**
     * 执行动作
     */
    public void executeAction(FlowSession session, ActionType actionType, FlowContext context) {
        // 1. 会话验证
        validateSession(session);

        // 2. 获取动作
        IFlowAction action = FlowActionFactory.getAction(actionType);

        // 3. 动作权限验证
        validateActionPermission(session, action);

        // 4. 执行动作
        action.execute(session, context);
    }

    private void validateSession(FlowSession session) {
        if (session == null) {
            throw new FlowSessionException("Session is null");
        }
        if (session.isExpired()) {
            throw new FlowSessionException("Session is expired");
        }
    }

    private void validateActionPermission(FlowSession session, IFlowAction action) {
        IFlowOperator operator = session.getOperator();
        if (!operator.hasPermission(action.getType())) {
            throw new FlowPermissionException("No permission to execute action: " + action.getType());
        }
    }
}
```

## FlowNodeManager

```java
package com.codingapi.flow.manager;

/**
 * 流程节点管理器
 * 负责获取流程节点信息和下一节点
 */
public class FlowNodeManager {

    /**
     * 获取下一节点
     */
    public IFlowNode getNextNode(IFlowNode currentNode, FlowContext context) {
        return switch (currentNode.getNodeType()) {
            case "Approval", "Handle" -> getNextByApprovalNode(currentNode, context);
            case "Router" -> getNextByRouterNode(currentNode, context);
            case "Parallel" -> getNextByParallelNode(currentNode, context);
            default -> getNextByDefault(currentNode, context);
        };
    }

    private IFlowNode getNextByApprovalNode(IFlowNode node, FlowContext context) {
        List<IFlowNode> nodes = context.getWorkflow().getNodes();
        return nodes.stream()
            .filter(n -> n.getOrder() > node.getOrder())
            .filter(IFlowNode::validate)
            .findFirst()
            .orElse(null);
    }

    private IFlowNode getNextByRouterNode(IFlowNode node, FlowContext context) {
        RouterNode routerNode = (RouterNode) node;
        return routerNode.evaluateCondition(context);
    }
}
```

## OperatorManager

```java
package com.codingapi.flow.manager;

/**
 * 操作者管理器
 * 负责管理流程操作者信息
 */
public class OperatorManager {

    private final FlowOperatorGateway operatorGateway;

    public OperatorManager(FlowOperatorGateway operatorGateway) {
        this.operatorGateway = operatorGateway;
    }

    /**
     * 获取操作者
     */
    public IFlowOperator getOperator(long operatorId) {
        IFlowOperator operator = operatorGateway.get(operatorId);
        if (operator == null) {
            throw new FlowNotFoundException("Operator not found: " + operatorId);
        }
        return operator;
    }

    /**
     * 批量获取操作者
     */
    public List<IFlowOperator> getOperators(List<Long> operatorIds) {
        return operatorGateway.findByIds(operatorIds);
    }

    /**
     * 获取流程管理员
     */
    public List<IFlowOperator> getFlowAdmins() {
        return operatorGateway.findByRole("FLOW_ADMIN");
    }

    /**
     * 匹配审批人
     */
    public List<IFlowOperator> matchApprovers(String matchScript, FlowContext context) {
        Object result = scriptRuntimeContext.executeScript(matchScript, context);
        if (result instanceof List) {
            return (List<IFlowOperator>) result;
        }
        return Collections.emptyList();
    }
}
```

## WorkflowStrategyManager

```java
package com.codingapi.flow.manager;

/**
 * 工作流策略管理器
 * 负责管理工作流级别的策略
 */
public class WorkflowStrategyManager {

    /**
     * 执行工作流策略
     */
    public void executeWorkflowStrategy(Workflow workflow, String strategyType, WorkflowContext context) {
        IWorkflowStrategy strategy = WorkflowStrategyFactory.getStrategy(strategyType);
        strategy.execute(workflow, context);
    }

    /**
     * 执行干预策略
     */
    public void executeInterfereStrategy(Workflow workflow, FlowRecord record, IFlowOperator operator) {
        WorkflowContext context = new WorkflowContext();
        context.setWorkflow(workflow);
        context.setRecord(record);
        context.setOperator(operator);

        IWorkflowStrategy strategy = WorkflowStrategyFactory.getStrategy(InterfereStrategy.TYPE);
        strategy.execute(workflow, context);
    }

    /**
     * 执行催办策略
     */
    public void executeUrgeStrategy(Workflow workflow, FlowRecord record) {
        WorkflowContext context = new WorkflowContext();
        context.setWorkflow(workflow);
        context.setRecord(record);

        IWorkflowStrategy strategy = WorkflowStrategyFactory.getStrategy(UrgeStrategy.TYPE);
        strategy.execute(workflow, context);
    }
}
```
