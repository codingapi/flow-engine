# 10 - 服务模块

## 模块概述

**包路径**: `com.codingapi.flow.service`

服务模块是业务逻辑的核心层，整合了节点、动作、策略、脚本等模块，提供完整的业务流程支持。

## FlowService

```java
package com.codingapi.flow.service;

/**
 * 流程服务主类
 * 整合所有流程操作，是对外提供服务的核心入口
 */
@Service
public class FlowService {

    private final FlowRecordService recordService;
    private final FlowActionService actionService;
    private final FlowNodeManager nodeManager;
    private final OperatorManager operatorManager;

    /**
     * 创建流程
     */
    public FlowRecord createFlow(FlowCreateRequest request) {
        // 1. 获取工作流定义
        Workflow workflow = workflowService.getWorkflow(request.getWorkflowId());

        // 2. 创建流程记录
        FlowRecord record = recordService.createRecord(workflow, request);

        // 3. 初始化第一个节点
        IFlowNode startNode = nodeManager.getStartNode(workflow);
        recordService.initFirstNode(record, startNode);

        // 4. 创建待办记录
        recordService.createTodoRecord(record, startNode);

        return record;
    }

    /**
     * 执行动作
     */
    public void executeAction(FlowActionRequest request) {
        FlowRecord record = recordService.getRecord(request.getRecordId());
        actionService.execute(record, request.getActionType(), request.getFormData());
        recordService.updateState(record);
    }

    /**
     * 获取流程详情
     */
    public FlowDetail getFlowDetail(Long recordId) {
        FlowRecord record = recordService.getRecord(recordId);
        IFlowNode currentNode = nodeManager.getCurrentNode(record);
        List<IFlowNode> nextNodes = nodeManager.getNextNodes(currentNode);

        return new FlowDetail(record, currentNode, nextNodes);
    }
}
```

## WorkflowService

```java
package com.codingapi.flow.service;

/**
 * 工作流服务
 */
@Service
public class WorkflowService {

    private final WorkflowRepository workflowRepository;
    private final WorkflowVersionRepository versionRepository;

    /**
     * 创建工作流
     */
    public Workflow createWorkflow(WorkflowCreateRequest request) {
        Workflow workflow = new WorkflowBuilder()
            .name(request.getName())
            .description(request.getDescription())
            .build();

        workflow = workflowRepository.save(workflow);

        // 创建初始版本
        WorkflowVersion version = new WorkflowVersion();
        version.setWorkflowId(workflow.getId());
        version.setVersion("1.0");
        version.setState(1);
        versionRepository.save(version);

        return workflow;
    }

    /**
     * 发布新版本
     */
    public WorkflowVersion publishVersion(Long workflowId, WorkflowVersionPublishRequest request) {
        Workflow workflow = workflowRepository.findById(workflowId);

        WorkflowVersion version = new WorkflowVersion();
        version.setWorkflowId(workflowId);
        version.setVersion(request.getVersion());
        version.setNodeMetas(request.getNodeMetas());
        version.setState(1);
        return versionRepository.save(version);
    }

    /**
     * 切换工作流版本
     */
    public void changeVersion(Long workflowId, Long versionId) {
        versionRepository.disableAll(workflowId);
        versionRepository.enable(versionId);
    }
}
```

## FlowProcessNodeService

```java
package com.codingapi.flow.service;

/**
 * 流程节点服务
 */
@Service
public class FlowProcessNodeService {

    private final FlowNodeManager nodeManager;
    private final WorkflowRepository workflowRepository;

    /**
     * 获取流程的当前节点
     */
    public ProcessNode getCurrentNode(Long recordId) {
        FlowRecord record = getRecord(recordId);
        Workflow workflow = workflowRepository.findById(record.getWorkflowId());
        IFlowNode node = nodeManager.getCurrentNode(record, workflow);

        return convertToProcessNode(node, record);
    }

    /**
     * 获取流程的下一可选节点
     */
    public List<ProcessNode> getNextNodes(Long recordId) {
        FlowRecord record = getRecord(recordId);
        Workflow workflow = workflowRepository.findById(record.getWorkflowId());
        IFlowNode currentNode = nodeManager.getCurrentNode(record, workflow);

        List<IFlowNode> nextNodes = nodeManager.getNextNodes(currentNode);
        return nextNodes.stream()
            .map(n -> convertToProcessNode(n, record))
            .collect(Collectors.toList());
    }

    /**
     * 获取节点的可执行动作列表
     */
    public List<NodeOption> getNodeActions(Long recordId) {
        FlowRecord record = getRecord(recordId);
        IFlowNode node = nodeManager.getCurrentNode(record);

        return node.getAvailableActions().stream()
            .map(action -> new NodeOption(action.getActionId(), action.getName(), action.getType()))
            .collect(Collectors.toList());
    }
}
```

## FlowRevokeService

```java
package com.codingapi.flow.service;

/**
 * 流程撤销服务
 */
@Service
public class FlowRevokeService {

    /**
     * 撤销流程
     */
    public void revoke(Long recordId, Long operatorId) {
        FlowRecord record = getRecord(recordId);

        // 1. 校验权限：只有发起人可以撤销
        if (!record.getInitOperatorId().equals(operatorId)) {
            throw new FlowPermissionException("Only initiator can revoke");
        }

        // 2. 校验状态：只能撤销待办状态的流程
        if (record.getState() != FlowRecord.STATE_TODO) {
            throw new FlowValidationException("Cannot revoke record in current state");
        }

        // 3. 执行撤销
        record.setState(FlowRecord.STATE_TERMINATED);
        recordService.save(record);

        // 4. 记录操作日志
        logService.saveRevokeLog(record, operatorId);
    }
}
```

## FlowUrgeService

```java
package com.codingapi.flow.service;

/**
 * 流程催办服务
 */
@Service
public class FlowUrgeService {

    /**
     * 催办流程
     */
    public void urge(Long recordId, Long operatorId) {
        FlowRecord record = getRecord(recordId);

        // 1. 检查催办间隔
        if (!urgeIntervalRepository.canUrge(recordId)) {
            throw new FlowValidationException("Urge interval not reached");
        }

        // 2. 发送催办通知
        notificationService.sendUrgeNotification(record);

        // 3. 记录催办日志
        urgeIntervalRepository.recordUrge(recordId, operatorId);
    }
}
```
